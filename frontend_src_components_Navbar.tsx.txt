import React from 'react';
import { Link, useNavigate } from 'react-router-dom';
import { Wifi } from 'lucide-react';

export default function Navbar() {
  const isAuthenticated = !!localStorage.getItem('authToken');
  const userRole = localStorage.getItem('userRole');
  const navigate = useNavigate();

  const handleLogout = () => {
    localStorage.removeItem('authToken');
    localStorage.removeItem('userRole');
    navigate('/');
  };

  return (
    <nav className="bg-white shadow-md">
      <div className="max-w-6xl mx-auto px-4 py-4 flex justify-between items-center">
        <Link to="/" className="flex items-center gap-2 text-2xl font-bold text-blue-600">
          <Wifi className="w-8 h-8" /> DataMart
        </Link>

        <div className="flex gap-4">
          {!isAuthenticated ? (
            <>
              <Link to="/auth" className="text-gray-700 font-bold hover:text-blue-600 transition">
                Sign In
              </Link>
            </>
          ) : (
            <>
              <Link to="/dashboard" className="text-gray-700 font-bold hover:text-blue-600 transition">
                Dashboard
              </Link>
              {userRole === 'admin' && (
                <Link to="/admin" className="text-gray-700 font-bold hover:text-blue-600 transition">
                  Admin
                </Link>
              )}
              <button
                onClick={handleLogout}
                className="text-gray-700 font-bold hover:text-red-600 transition"
              >
                Logout
              </button>
            </>
          )}
        </div>
      </div>
    </nav>
  );
}
